Leetcode 215 数组中的第 K 个最大的元素 直接排序后返回 k 即可。

class Solution {
public:
int findKthLargest(vector<int>& nums, int k) {
	std::sort(nums.begin(),nums.end());
	return nums[nums.size()-k];
	}
};
